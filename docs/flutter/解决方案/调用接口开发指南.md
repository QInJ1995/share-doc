# 调用接口开发指南

## 创建dao文件夹

在lib文件夹下创建一个dao文件夹，用于存放接口请求的类。

## 创建接口请求类

在dao文件夹下创建一个接口请求类，例如`UserDao.dart`。

```dart
import 'package:dio/dio.dart';
import 'package:flutter/material.dart';
import 'package:myapp/models/user.dart';

class UserDao {
  static Future<User> login(String username, String password) async {
    try {
      Response response = await Dio().post(
        'http://example.com/api/login',
        data: {'username': username, 'password': password},
      );
      return User.fromJson(response.data);
    } catch (e) {
      throw Exception('登录失败');
    }
  }
}
```

## 在页面中使用接口请求类

在页面中使用接口请求类，例如`LoginPage.dart`。

```dart
import 'package:flutter/material.dart';
import 'package:myapp/dao/UserDao.dart';
import 'package:myapp/models/user.dart';

class LoginPage extends StatefulWidget {
  @override
  _LoginPageState createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
  final TextEditingController _usernameController = TextEditingController();
  final TextEditingController _passwordController = TextEditingController();

  Future<User> _login() async {
    String username = _usernameController.text;
    String password = _passwordController.text;
    return await UserDao.login(username, password);
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('登录'),
      ),
      body: Padding(
        padding: EdgeInsets.all(16.0),
        child: Column(
          children: [
            TextField(
              controller: _usernameController,
              decoration: InputDecoration(
                labelText: '用户名',
              ),
            ),
            TextField(
              controller: _passwordController,
              decoration: InputDecoration(
                labelText: '密码',
              ),
              obscureText: true,
            ),
            SizedBox(height: 16.0),
            ElevatedButton(
              onPressed: () async {
                User user = await _login();
                // 处理登录成功后的逻辑
              },
              child: Text('登录'),
            ),
          ],
        ),
      ),
    );
  }
}
```

## 处理接口请求结果

在页面中处理接口请求结果，例如`LoginPage.dart`。

```dart
import 'package:flutter/material.dart';
import 'package:myapp/dao/UserDao.dart';
import 'package:myapp/models/user.dart';

class LoginPage extends StatefulWidget {
```

## 处理接口请求超时

在页面中处理接口请求超时，例如`LoginPage.dart`。

```dart
import 'package:flutter/material.dart';
import 'package:myapp/dao/UserDao.dart';
import 'package:myapp/models/user.dart';

class LoginPage extends StatefulWidget {
  @override
  _LoginPageState createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
  final TextEditingController _usernameController = TextEditingController();
  final TextEditingController _passwordController = TextEditingController();
  bool _loading = false;

  Future<User> _login() async {
    setState(() {
      _loading = true;
    });
    String username = _usernameController.text;
    String password = _passwordController.text;
    try {
      User user = await UserDao.login(username, password);
      setState(() {
        _loading = false;
      });
      // 处理登录成功后的逻辑
      return user;
    } catch (e) {
      setState(() {
        _loading = false;
      });
      // 处理登录失败后的逻辑
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('登录失败')),
      );
      throw Exception('登录失败');
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('登录'),
      ),
      body: Padding(
        padding: EdgeInsets.all(16.0),
        child: Column(
          children: [
            TextField(
              controller: _usernameController,
              decoration: InputDecoration(
                labelText: '用户名',
              ),
            ),
            TextField(
              controller: _passwordController,
              decoration: InputDecoration(
                labelText: '密码',
              ),
              obscureText: true,
            ),
            SizedBox(height: 16.0),
            ElevatedButton(
              onPressed: _loading ? null : () async {
                try {
                  User user = await _login();
                  // 处理登录成功后的逻辑
                } catch (e) {
                  if (e is TimeoutException) {
                    // 处理接口请求超时
                    ScaffoldMessenger.of(context).showSnackBar(
                      SnackBar(content: Text('接口请求超时')),
                    );
                  } else {
                    // 处理其他异常
                    ScaffoldMessenger.of(context).showSnackBar(
                      SnackBar(content: Text('登录失败')),
                    );
                  }
                }
              },
              child: _loading ? CircularProgressIndicator() : Text('登录'),
            ),
          ],
        ),
      ),
    );
  }
}
```
